# Test 03: Наследование методов

class Animal is
  method bark() : Integer is
    return (1)
  end
end

class Dog extends Animal is
  method bark() : Integer is
    return (2)
  end
end

class Program is
  var dog : Dog()
  var result : Integer()

  var animal : Animal()
  var result2 : Integer()
  
  method main() is
    result := dog.bark()
    print(result)

    result2 := animal.bark()
    print(result2)
  end
end
