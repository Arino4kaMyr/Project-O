# Test 06: Перегрузка методов

class Calculator is
  method add(x: Integer) : Integer is
    return x
  end
  
  method add(x: Integer, y: Integer) : Integer is
    return x.Plus(y)
  end
end

class Program is
  var calc : Calculator()
  var x : Integer(5)
  var y : Integer(10)

  var res1 : Integer()
  var res2 : Integer()
  
  method main() is
    res1 := calc.add(x)
    res2 := calc.add(x, y)

    print(res1)
    print(res2)
  end
end
